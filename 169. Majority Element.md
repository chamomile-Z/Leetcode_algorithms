# My Solution01 (dictionary)
```Python
def majorityElement(self, nums: List[int]) -> int:
  d = {}

  for num in nums:
    if num in d:
      d[num] += 1
    else:
      d[num] = 1

  v = list(d.values())
  k = list(d.keys())

  return k[v.index(max(v))]
```

## Method
- Time Complexity: O(n)
- Space Complexity: O(n)

```Python
def majorityElement(self, nums: List[int]) -> int:
  d = {}
  l = len(nums)
      
  for num in nums:
    if num in d:
      d[num] += 1
      if d[num] > (l // 2):
        return num
      else:
        d[num] = 1
        if len(nums) == 1:
          return num
```

## Method
- Time Complexity: O(n)
- Space Complexity: O(n)
